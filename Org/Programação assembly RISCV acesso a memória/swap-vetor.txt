2) Implemente uma função chamada "swap_vetor" que recebe o endereço inicial de um vetor e dois índices (nos registradores  a0, a1  e a2 respectivamente). A função deve realizar a troca dos valores presentes nos dois indices recebidos. Exemplo: Considerando o vetor armazendando os valores 2, 3, 1, -4, 6. Ao receber os indices 0 e 3 a função deve trocar de posição os valores 2 e -4. O valor final do vetor será: -4, 3, 1, 2, 6.

---------------------------------------------------------------------------------------

.data
vetor: .word	-4, 3, 1, 2, 6

.text
main:
	la	a0, vetor		#carrega o vetor em a0
	addi	a1, zero, 0		#seta o índice zero em a1
	addi	a2, zero, 3		#seta o índice 3 em a2
	addi	a3, zero, 5		#tamanho do vetor em a0
	addi	a4, zero, 0		#controle do laço
	jal	swap_vetor
	nop
	ebreak
swap_vetor:
	beq	a4, a3, fim_laco	#verifica se o laço acabou
	beq	a4, a1, salvaValor_a1	#verifica se o indice do a4=a1
	j 	verificaValor_a2
salvaValor_a1:
	lw	t1, 0(a0)		#grava o valor da posição do vetor em t1
	addi	a4, a4, 1		#incrementa índice do vetor
	addi	a5, a0, 0		#salva endereço do índice do a1
	addi	a0, a0, 4		#incrementa o endereço do vetor
	j	swap_vetor
verificaValor_a2:
	beq	a4, a2, salvaValor_a2
	addi	a0, a0, 4		#incrementa o endereço do vetor
	addi	a4, a4, 1		#incrementa índice do vetor
	j	swap_vetor
salvaValor_a2:
	addi	a6, a0, 0		#salva endereço do índice do a2
	lw	t2, 0(a0)		#grava o valor da posição do vetor em t2
	addi	a0, a6, 0		#seta a0 para o endereço do índice de a2
	sw	t1, 0(a0)		#escreve o valor do índice a2 em a1
	lw	t0, 0(a0)
	addi	a0, a5, 0		#seta a0 para o endereço do índice de a1
	sw	t2, 0(a0)		#escreve o valor do índice a2 em a1
	lw	s1, 0(a0)
	j	fim_laco
fim_laco:
	ret
